-- Step 1: Create all tables without constraints
CREATE TABLE parent_company (
  ParentCoID int PRIMARY KEY
);

CREATE TABLE product (
  ASIN varchar(10) PRIMARY KEY,
  ProductName varchar(45),
  Weight decimal(10,2),
  Price decimal(10,2),
  Count int,
  TimeStamp timestamp
);

CREATE TABLE customer (
  CustomerID varchar(16) PRIMARY KEY,
  Username varchar(16) NOT NULL,
  Email varchar(255),
  Password varchar(32) NOT NULL,
  UserType varchar(10),
  CreateTime timestamp DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE shrinkflation_event (
  EventID varchar(16) PRIMARY KEY,
  DetectedDate timestamp DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE shrinkflation_metric (
  ShrinkflationMetricID varchar(16) PRIMARY KEY,
  ChangeWeight decimal(10,2),
  ChangeCount int,
  ChangePrice decimal(10,2),
  PercentChangePrice decimal(5,2),
  PercentChangeCount decimal(5,2),
  PercentChangeWeight decimal(5,2)
);

CREATE TABLE variant (
  VariantID int PRIMARY KEY,
  ChildASIN varchar(10)
);

CREATE TABLE sector (
  SectorID int PRIMARY KEY
);

CREATE TABLE seller (
  SellerID varchar(16) PRIMARY KEY,
  SellerFName varchar(16) NOT NULL,
  SellerLName varchar(16) NOT NULL,
  SellerType varchar(16)
);

CREATE TABLE subscription (
  SubscriptionID varchar(16) PRIMARY KEY,
  CustomerID varchar(16) NOT NULL,
  Features varchar(16) NOT NULL,
  SubscriptionLevel varchar(16) NOT NULL,
  SubscriptionPrice decimal(10,2) NOT NULL,
  SubscriptionDate date
);

CREATE TABLE cohort (
  CohortID int PRIMARY KEY
);

CREATE TABLE trend (
  TrendID varchar(16) PRIMARY KEY,
  ShrinkflationMetricID varchar(16) NOT NULL,
  ShrinkflationIndex decimal(10,2),
  StartDate date,
  EndDate date
);

CREATE TABLE triggers (
  TrendID varchar(16) NOT NULL,
  EventID varchar(16) NOT NULL,
  PRIMARY KEY (TrendID, EventID)
);

CREATE TABLE produces (
  ASIN varchar(10) PRIMARY KEY,
  ParentCOID int
);

CREATE TABLE calculates (
  ASIN varchar(10) NOT NULL,
  ShrinkflationMetricID varchar(16) NOT NULL,
  CurWeight decimal(10,2) NOT NULL,
  CurCount int NOT NULL,
  CurPrice decimal(10,2) NOT NULL,
  PRIMARY KEY (ASIN, ShrinkflationMetricID)
);

CREATE TABLE has_variant (
  ASIN varchar(10) NOT NULL,
  VariantID int NOT NULL,
  PRIMARY KEY (ASIN, VariantID)
);

CREATE TABLE access (
  CustomerID varchar(16) NOT NULL,
  EventID varchar(16) NOT NULL,
  PRIMARY KEY (CustomerID, EventID)
);

CREATE TABLE is_also_a (
  CustomerID varchar(16) NOT NULL,
  SellerID varchar(16) NOT NULL,
  PRIMARY KEY (CustomerID, SellerID)
);

CREATE TABLE notification (
  NotificationID varchar(16) PRIMARY KEY,
  EventID varchar(16) NOT NULL,
  MessageContent varchar(250),
  MessageTime timestamp DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE notify (
  SubscriptionID varchar(16) NOT NULL,
  NotificationID varchar(16) NOT NULL,
  PRIMARY KEY (SubscriptionID, NotificationID)
);

CREATE TABLE supplies (
  SellerID varchar(16) NOT NULL,
  ParentCOID int NOT NULL,
  PRIMARY KEY (SellerID, ParentCOID)
);

CREATE TABLE competes_in (
  ParentCoID int NOT NULL,
  SectorID int NOT NULL,
  PRIMARY KEY (ParentCoID, SectorID)
);

CREATE TABLE competes_with (
  SectorID int NOT NULL,
  CohortID int NOT NULL,
  PRIMARY KEY (SectorID, CohortID)
);

CREATE TABLE sells (
  Sells_SellerID varchar(16) NOT NULL,
  Sells_ASIN varchar(45) NOT NULL,
  PRIMARY KEY (Sells_SellerID, Sells_ASIN)
);

-- Step 2: Add Foreign Key Constraints
ALTER TABLE produces
  ADD CONSTRAINT produces_fk_parent_company FOREIGN KEY (ParentCOID) REFERENCES parent_company (ParentCoID);

ALTER TABLE produces
  ADD CONSTRAINT produces_fk_product FOREIGN KEY (ASIN) REFERENCES product (ASIN);

ALTER TABLE calculates
  ADD CONSTRAINT calculates_fk_product FOREIGN KEY (ASIN) REFERENCES product (ASIN),
  ADD CONSTRAINT calculates_fk_metric FOREIGN KEY (ShrinkflationMetricID) REFERENCES shrinkflation_metric (ShrinkflationMetricID);

ALTER TABLE has_variant
  ADD CONSTRAINT has_variant_fk_product FOREIGN KEY (ASIN) REFERENCES product (ASIN),
  ADD CONSTRAINT has_variant_fk_variant FOREIGN KEY (VariantID) REFERENCES variant (VariantID);

ALTER TABLE access
  ADD CONSTRAINT access_fk_customer FOREIGN KEY (CustomerID) REFERENCES customer (CustomerID),
  ADD CONSTRAINT access_fk_event FOREIGN KEY (EventID) REFERENCES shrinkflation_event (EventID);

ALTER TABLE is_also_a
  ADD CONSTRAINT is_also_a_fk_customer FOREIGN KEY (CustomerID) REFERENCES customer (CustomerID),
  ADD CONSTRAINT is_also_a_fk_seller FOREIGN KEY (SellerID) REFERENCES seller (SellerID);

ALTER TABLE notification
  ADD CONSTRAINT notification_fk_event FOREIGN KEY (EventID) REFERENCES shrinkflation_event (EventID);

ALTER TABLE notify
  ADD CONSTRAINT notify_fk_subscription FOREIGN KEY (SubscriptionID) REFERENCES subscription (SubscriptionID),
  ADD CONSTRAINT notify_fk_notification FOREIGN KEY (NotificationID) REFERENCES notification (NotificationID);

ALTER TABLE competes_in
  ADD CONSTRAINT competes_in_fk_sector FOREIGN KEY (SectorID) REFERENCES sector (SectorID),
  ADD CONSTRAINT competes_in_fk_parent_company FOREIGN KEY (ParentCoID) REFERENCES parent_company (ParentCoID);

ALTER TABLE competes_with
  ADD CONSTRAINT competes_with_fk_sector FOREIGN KEY (SectorID) REFERENCES sector (SectorID),
  ADD CONSTRAINT competes_with_fk_cohort FOREIGN KEY (CohortID) REFERENCES cohort (CohortID);

ALTER TABLE sells
  ADD CONSTRAINT sells_fk_seller FOREIGN KEY (Sells_SellerID) REFERENCES seller (SellerID),
  ADD CONSTRAINT sells_fk_product FOREIGN KEY (Sells_ASIN) REFERENCES product (ASIN);

ALTER TABLE supplies
  ADD CONSTRAINT supplies_fk_seller FOREIGN KEY (SellerID) REFERENCES seller (SellerID),
  ADD CONSTRAINT supplies_fk_parent_company FOREIGN KEY (ParentCOID) REFERENCES parent_company (ParentCoID);

ALTER TABLE triggers
  ADD CONSTRAINT triggers_fk_trend FOREIGN KEY (TrendID) REFERENCES trend (TrendID),
  ADD CONSTRAINT triggers_fk_event FOREIGN KEY (EventID) REFERENCES shrinkflation_event (EventID);

ALTER TABLE subscription
  ADD CONSTRAINT subscription_fk_customer FOREIGN KEY (CustomerID) REFERENCES customer (CustomerID);

ALTER TABLE trend
  ADD CONSTRAINT trend_fk_metric FOREIGN KEY (ShrinkflationMetricID) REFERENCES shrinkflation_metric (ShrinkflationMetricID);
